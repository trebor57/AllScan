# Useful aliases for AllStar
# Add these to .bashrc by running 'nano ~/.bashrc', then copy in the below,
#  Ctrl-S to Save, Ctrl-x to exit, then run 'source ~/.bashrc'

alias la='ls -A'
alias l='ls -CF'
alias www='cd /var/www/html'
alias ast='cd /etc/asterisk'
alias d='ls -lart'

alias gr='php -q ~/wgrep.php -r'
alias gv='grep -v'
alias gci='php -q ~/cgrep.php'

alias aststop='sudo service asterisk stop'

alias astver='sudo asl-show-version'

alias makeasl='cd ~/asl3-asterisk-*; make'
